#include <stdio.h>
#include <string.h>


char prod[10][10];
char firstSet[10][10];
int n;


void findFirst(char *res, char ch) {
    for (int i = 0; i < n; i++) {
        if (prod[i][0] == ch) {
            if (!isupper(prod[i][2])) {
                strncat(res, &prod[i][2], 1);
            } else {
                findFirst(res, prod[i][2]);
            }
        }
    }
}


int main() {
    char res[10];
    printf("Enter number of productions: ");
    scanf("%d", &n);
    printf("Enter productions (like E=E+T or E=@):\n");
    for (int i = 0; i < n; i++) {
        scanf("%s", prod[i]);
    }


    for (int i = 0; i < n; i++) {
        strcpy(res, "");
        findFirst(res, prod[i][0]);
        printf("First(%c) = { ", prod[i][0]);
        for (int j = 0; j < strlen(res); j++)
            printf("%c ", res[j]);
        printf("}\n");
    }


    return 0;
}

✅ 4. FOLLOW Set Calculation
c
CopyEdit
#include <stdio.h>
#include <string.h>
#include <stdbool.h>


char productions[10][10];
int n;


void follow(char c, char *res) {
    if (c == productions[0][0])
        strcat(res, "$");


    for (int i = 0; i < n; i++) {
        for (int j = 2; j < strlen(productions[i]); j++) {
            if (productions[i][j] == c) {
                if (productions[i][j+1] != '\0' && !isupper(productions[i][j+1])) {
                    strncat(res, &productions[i][j+1], 1);
                } else if (productions[i][j+1] == '\0') {
                    if (c != productions[i][0])
                        follow(productions[i][0], res);
                }
            }
        }
    }
}


int main() {
    printf("Enter number of productions: ");
    scanf("%d", &n);
    printf("Enter productions (like A=aB):\n");
    for (int i = 0; i < n; i++) {
        scanf("%s", productions[i]);
    }


    for (int i = 0; i < n; i++) {
        char res[20] = "";
        follow(productions[i][0], res);
        printf("Follow(%c) = { ", productions[i][0]);
        for (int j = 0; j < strlen(res); j++)
            printf("%c ", res[j]);
        printf("}\n");
    }


    return 0;
}

#include <stdio.h>
#include <stdlib.h>


int main() {
    FILE *fp;
    char filename[] = "example_program.bin";
    char ch;


    fp = fopen(filename, "rb");
    if (fp == NULL) {
        printf("Error: File not found.\n");
        return 1;
    }


    printf("Loader: Executing file \"%s\"...\n", filename);
    printf("Loaded content (binary data):\n");


    while ((ch = fgetc(fp)) != EOF) {
        printf("%02X ", (unsigned char)ch);
    }


    fclose(fp);
    return 0;
}
